# 方法 1：开发模式安装（推荐）

## 1. 进入库的根目录（含setup.py/pyproject.toml）
cd /path/to/your/local_lib

## 2. 激活目标conda环境
conda activate your_env

## 3. 开发模式安装（-e 参数创建符号链接）
pip install -e . 

# 方法 2：直接添加至 Python 路径

## 临时生效（仅当前会话）
export PYTHONPATH="/path/to/your/local_lib:$PYTHONPATH"

## 永久生效（添加到conda环境变量）
echo 'export PYTHONPATH="/path/to/your/local_lib:$PYTHONPATH"' >> $CONDA_PREFIX/etc/conda/activate.d/env_vars.sh
